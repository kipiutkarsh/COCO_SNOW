CREATE OR REPLACE FUNCTION SERVICENOW_DEST_DB.SERVICENOW_DEST_SCHEMA.GET_SIMILAR_INCIDENTS(
    incident_description VARCHAR,
    incident_category VARCHAR,
    --incident_subcategory VARCHAR,
    top_k INTEGER
)
RETURNS TABLE (
    incident_number VARCHAR,
    description VARCHAR,
    category VARCHAR,
    --subcategory VARCHAR,
    assignment_group VARCHAR,
    assigned_to VARCHAR,
    similarity_score FLOAT
)
AS
$$
    WITH incident_embedding AS (
        SELECT SNOWFLAKE.CORTEX.EMBED_TEXT_768(
            'e5-base-v2',
            COALESCE(incident_description, '') || ' ' || 
            COALESCE(incident_category, '') 
        ) AS query_vector
    )
    SELECT 
        t.NUMBER AS incident_number,
        t.SHORT_DESCRIPTION AS description,
        t.CATEGORY AS category,
        --t.SUBCATEGORY AS subcategory,
        t.ASSIGNMENT_GROUP AS assignment_group,
        t.ASSIGNED_TO AS assigned_to,
        VECTOR_COSINE_SIMILARITY(t.DESCRIPTION_EMBEDDING, ie.query_vector) AS similarity_score
    FROM 
        SERVICENOW_DEST_DB.SERVICENOW_DEST_SCHEMA.INCIDENT_TRAINING_EXAMPLES t,
        incident_embedding ie
    WHERE 
        t.DESCRIPTION_EMBEDDING IS NOT NULL
        AND t.STATE = 'Resolved'
        AND t.ASSIGNMENT_GROUP IS NOT NULL
        AND t.ASSIGNED_TO IS NOT NULL
    ORDER BY 
        similarity_score DESC
    LIMIT top_k
$$;
